// FICHIER: backend/__tests__/unit/auth.middleware.test.js
// ============================================================
// Tests unitaires pour le middleware d'authentification
// On mocke les dépendances externes (argon2, jwt, models)

const jwt = require("jsonwebtoken");
const argon2 = require("argon2");

// Mock des modules avant de les importer
jest.mock("argon2");
jest.mock("jsonwebtoken");
jest.mock("../../src/models", () => ({
  tokenBlacklist: {
    findByToken: jest.fn(),
    insert: jest.fn(),
  },
}));

const models = require("../../src/models");
const {
  hashPassword,
  verifyPassword,
  verifyToken,
} = require("../../src/middleware/auth");

// Helper : crée des objets req/res/next mockés pour chaque test
const createMockReqResNext = (overrides = {}) => {
  const req = {
    body: {},
    headers: {},
    user: null,
    User_ID: null,
    ...overrides,
  };
  const res = {
    status: jest.fn().mockReturnThis(),
    send: jest.fn().mockReturnThis(),
    sendStatus: jest.fn().mockReturnThis(),
    json: jest.fn().mockReturnThis(),
  };
  const next = jest.fn();
  return { req, res, next };
};

describe("Auth Middleware", () => {
  afterEach(() => {
    jest.clearAllMocks();
  });

  // -------------------------------------------------------
  // hashPassword
  // -------------------------------------------------------
  describe("hashPassword", () => {
    /**
     * TEST AUTH-11 : hashPassword doit hasher le mot de passe avec argon2
     * et stocker le résultat dans req.body.hashedPassword,
     * puis supprimer req.body.Password et appeler next()
     */
    it("should hash the password and call next()", async () => {
      const { req, res, next } = createMockReqResNext({
        body: { Password: "MySecretPass123" },
      });

      argon2.hash.mockResolvedValue("$argon2id$hashedvalue");

      await hashPassword(req, res, next);

      // Le mot de passe en clair doit être supprimé
      expect(req.body.Password).toBeUndefined();
      // Le hash doit être stocké
      expect(req.body.hashedPassword).toBe("$argon2id$hashedvalue");
      // argon2.hash doit avoir été appelé avec le bon mot de passe
      expect(argon2.hash).toHaveBeenCalledWith(
        "MySecretPass123",
        expect.objectContaining({
          type: argon2.argon2id,
          memoryCost: 65536,
          timeCost: 5,
          parallelism: 1,
        })
      );
      // next() doit être appelé pour passer au middleware suivant
      expect(next).toHaveBeenCalled();
    });

    /**
     * TEST: hashPassword doit retourner 500 si argon2 échoue
     */
    it("should return 500 if argon2.hash throws", async () => {
      const { req, res, next } = createMockReqResNext({
        body: { Password: "test" },
      });

      argon2.hash.mockRejectedValue(new Error("Hashing failed"));

      await hashPassword(req, res, next);

      expect(res.sendStatus).toHaveBeenCalledWith(500);
      expect(next).not.toHaveBeenCalled();
    });
  });

  // -------------------------------------------------------
  // verifyPassword
  // -------------------------------------------------------
  describe("verifyPassword", () => {
    /**
     * TEST AUTH-01 : verifyPassword avec mot de passe correct
     * doit retourner 200 avec un token JWT et les infos user
     */
    it("should return 200 with token when password is correct", async () => {
      const mockUser = {
        User_ID: 5,
        Username: "testuser",
        hashedPassword: "$argon2id$existinghash",
      };
      const { req, res } = createMockReqResNext({
        body: { Password: "correctPassword" },
        user: mockUser,
      });

      argon2.verify.mockResolvedValue(true);
      jwt.sign.mockReturnValue("fake.jwt.token");

      // Sauvegarde de la variable d'env originale
      const originalSecret = process.env.JWT_SECRET;
      process.env.JWT_SECRET = "test-secret";

      await verifyPassword(req, res);

      // Vérifie qu'argon2.verify est appelé avec le bon hash et mot de passe
      expect(argon2.verify).toHaveBeenCalledWith(
        "$argon2id$existinghash",
        "correctPassword"
      );
      // Vérifie la génération du JWT
      expect(jwt.sign).toHaveBeenCalledWith(
        { sub: 5 },
        "test-secret",
        { expiresIn: "4h" }
      );
      // Vérifie la réponse
      expect(res.status).toHaveBeenCalledWith(200);
      expect(res.send).toHaveBeenCalledWith(
        expect.objectContaining({
          authToken: "fake.jwt.token",
          user: mockUser,
          message: "Login successful",
        })
      );

      process.env.JWT_SECRET = originalSecret;
    });

    /**
     * TEST AUTH-03 : verifyPassword avec mot de passe incorrect
     * doit retourner 401
     */
    it("should return 401 when password is incorrect", async () => {
      const { req, res } = createMockReqResNext({
        body: { Password: "wrongPassword" },
        user: {
          User_ID: 5,
          hashedPassword: "$argon2id$existinghash",
        },
      });

      argon2.verify.mockResolvedValue(false);

      await verifyPassword(req, res);

      expect(res.status).toHaveBeenCalledWith(401);
      expect(res.send).toHaveBeenCalledWith("Incorrect password");
    });

    /**
     * TEST AUTH-04 : verifyPassword sans user dans req
     * doit retourner 500 (pas de crash)
     */
    it("should return 500 when req.user is undefined", async () => {
      const { req, res } = createMockReqResNext({
        body: { Password: "anyPassword" },
        user: null,
      });

      await verifyPassword(req, res);

      expect(res.sendStatus).toHaveBeenCalledWith(500);
    });

    /**
     * TEST: verifyPassword quand req.user existe mais sans hashedPassword
     */
    it("should return 500 when req.user has no hashedPassword", async () => {
      const { req, res } = createMockReqResNext({
        body: { Password: "anyPassword" },
        user: { User_ID: 5 }, // pas de hashedPassword
      });

      await verifyPassword(req, res);

      expect(res.sendStatus).toHaveBeenCalledWith(500);
    });
  });

  // -------------------------------------------------------
  // verifyToken
  // -------------------------------------------------------
  describe("verifyToken", () => {
    /**
     * TEST AUTH-07 : Pas de header Authorization → 401
     */
    it("should return 401 when no Authorization header is present", async () => {
      const { req, res, next } = createMockReqResNext({
        headers: {},
      });

      await verifyToken(req, res, next);

      expect(res.status).toHaveBeenCalledWith(401);
      expect(next).not.toHaveBeenCalled();
    });

    /**
     * TEST AUTH-10 : Authorization header sans "Bearer " → 401
     */
    it("should return 401 when Authorization header does not start with Bearer", async () => {
      const { req, res, next } = createMockReqResNext({
        headers: { authorization: "Basic sometoken" },
      });

      await verifyToken(req, res, next);

      expect(res.status).toHaveBeenCalledWith(401);
      expect(next).not.toHaveBeenCalled();
    });

    /**
     * TEST AUTH-06 : Token blacklisté → 401
     */
    it("should return 401 when token is blacklisted", async () => {
      const { req, res, next } = createMockReqResNext({
        headers: { authorization: "Bearer blacklisted.token.here" },
      });

      // Le token est trouvé dans la blacklist
      models.tokenBlacklist.findByToken.mockResolvedValue([
        [{ jwtToken: "blacklisted.token.here" }],
      ]);

      await verifyToken(req, res, next);

      expect(res.status).toHaveBeenCalledWith(401);
      expect(res.send).toHaveBeenCalledWith(
        "Session Expired. Please log in again."
      );
      expect(next).not.toHaveBeenCalled();
    });

    /**
     * TEST: Token valide → next() est appelé avec User_ID
     */
    it("should call next() and set req.User_ID when token is valid", async () => {
      const { req, res, next } = createMockReqResNext({
        headers: { authorization: "Bearer valid.token.here" },
      });

      // Token pas dans la blacklist
      models.tokenBlacklist.findByToken.mockResolvedValue([[]]);

      // jwt.verify appelle le callback avec decoded
      jwt.verify.mockImplementation((token, secret, callback) => {
        callback(null, { sub: 42 });
      });

      process.env.JWT_SECRET = "test-secret";

      await verifyToken(req, res, next);

      expect(req.User_ID).toBe(42);
      expect(next).toHaveBeenCalled();
    });

    /**
     * TEST AUTH-08/AUTH-09 : Token invalide/expiré → 401
     */
    it("should return 401 when jwt.verify fails (expired or invalid)", async () => {
      const { req, res, next } = createMockReqResNext({
        headers: { authorization: "Bearer expired.token.here" },
      });

      models.tokenBlacklist.findByToken.mockResolvedValue([[]]);

      jwt.verify.mockImplementation((token, secret, callback) => {
        callback(new Error("jwt expired"), null);
      });

      await verifyToken(req, res, next);

      expect(res.sendStatus).toHaveBeenCalledWith(401);
      expect(next).not.toHaveBeenCalled();
    });
  });
});
